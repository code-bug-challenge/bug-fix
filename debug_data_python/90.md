### 修改方案：


```python
a, b = map(int, input('输入区间a,b:').split(','))
if a > b:
    a, b = b, a
for i in range(a, b + 1):
    s = 0  # 循环次数
    c = i
    while True:
        n = 0
        for j in str(c):  # 遍历每个字符
            n += int(j) ** 2
        c = n
        s += 1
        if s > 200 or n == i:  # 当计算次数>200或等于i,退出循环
            break
        if n == 1:
            print(i, s)
            break
print('结束')

```
