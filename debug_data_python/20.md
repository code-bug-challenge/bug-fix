def ResultWriteExcel(filename, resultData):

   # 写入一:写到表中
    wb = load_workbook(filename)  # 生成一个已存在的wookbook对象
    wb1 = wb["签到Sheet"]  #  工作簿名
    # wb1 = wb.active#激活sheet
