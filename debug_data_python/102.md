    cReturn = ""

        r.raise_for_status()
        r.encoding = r.apparent_encoding
        cReturn = r.text
    except:
        cReturn = ''
    return cReturn


def download(List):
    for u in List:
        try:
            path=root+u.split('/')[-1]

            r = requests.get(u, timeout=30)
