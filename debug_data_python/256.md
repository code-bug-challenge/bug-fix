### 修改方案：
改成这样，如有帮助望采纳：

```python
n = 2
number = 0
for i in range(1, n + 1):
    for b in range(n + 1 - i):
        print(' ', end='')
    n_list = []
    for c in range(i):
        n_list.append(str(number))
        number += 1
    print(' '.join(n_list))
for a in range(n - 1, 0, -1):
    for j in range(0, n - a):
        print(end='  ')
    k_list = []
    for k in range(0, a):
        k_list.append(str(number))
    print(' '.join(k_list))






```
